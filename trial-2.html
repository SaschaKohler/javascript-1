Multiply two variables prompted by the user

<script>
"use strict";
let var1 = prompt("Please give me a number: ");
let var2 = prompt("Here we go for a second number: ");
let result = Number(var1 * var2);
document.write(var1 + " * " + var2 + " = " + result);
</script>
