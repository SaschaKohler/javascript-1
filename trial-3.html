Challenge 3

Ask for the password

<script>
"use strict";
const username = "admin";
const password = "123456";
let prompted_user = prompt("Please give me your username: ");
let prompted_pass = prompt("Type in the correct password: ");
if (prompted_user == username) {
document.writeln("the given username is correct");
} else if (prompted_user != username) {
document.writeln("Please provide a valid username!");
}
if (prompted_pass == password) {
document.writeln("the given password is correct (wasn't to complicated,he)");
} else if (prompted_pass != password) {
document.writeln("Sorry..wrong password");
}
</script>
