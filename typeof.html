Datentypen ermitteln mit typeof

<script>
"use strict";
let integer = 3;
let float = 2.34;
let char = "a";
let word = "hello";
let boolean = true;
let dummy;
document.write("Integer: " + typeof integer + " <br>");
document.write("Float: " + typeof float + " <br>");
document.write("Char: " + typeof char + " <br>");
document.write("Word: " + typeof word + " <br>");
document.write("dummy: " + typeof dummy + " <br>");
</script>
